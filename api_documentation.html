多模态对话系统

Python - 统一接口封装 
import requests
import json
from typing import Dict, Any, Optional

class UnifiedAI:
    def __init__(self, api_key: str, base_url: str = "https://api.example.com"):
        self.api_key = api_key
        self.base_url = base_url
        self.headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }
    
    def chat(self, 
             message: str, 
             model: str = "gpt-4",
             images: Optional[list] = None,
             audio: Optional[str] = None,
             stream: bool = False) -> Dict[str, Any]:
        """统一的聊天接口，支持文本、图像、音频"""
        
        content = [{"type": "text", "text": message}]
        
        # 添加图像
        if images:
            for img in images:
                content.append({
                    "type": "image_url",
                    "image_url": {"url": img}
                })
        
        # 添加音频
        if audio:
            content.append({
                "type": "audio",
                "audio": {"data": audio}
            })
        
        payload = {
            "model": model,
            "messages": [{"role": "user", "content": content}],
            "stream": stream
        }
        
        response = requests.post(
            f"{self.base_url}/v1/chat/completions",
            headers=self.headers,
            json=payload,
            stream=stream
        )
        
        if stream:
            return response
        else:
            return response.json()
    
    def transcribe_audio(self, audio_file: str) -> str:
        """语音转文字"""
        with open(audio_file, "rb") as f:
            files = {"file": f}
            response = requests.post(
                f"{self.base_url}/v1/audio/transcriptions",
                headers={"Authorization": f"Bearer {self.api_key}"},
                files=files,
                data={"model": "whisper-1"}
            )
        return response.json()["text"]
    
    def text_to_speech(self, text: str, voice: str = "alloy") -> bytes:
        """文字转语音"""
        response = requests.post(
            f"{self.base_url}/v1/audio/speech",
            headers=self.headers,
            json={
                "model": "tts-1",
                "voice": voice,
                "input": text
            }
        )
        return response.content

# 使用示例
ai = UnifiedAI("your_api_key")

# 文本对话
response = ai.chat("你好，请介绍一下你自己")
print(response['choices'][0]['message']['content'])

# 多模态对话
response = ai.chat(
    "这张图片里有什么？",
    images=["https://example.com/image.jpg"]
)

# 语音转文字
transcript = ai.transcribe_audio("audio.mp3")
print(f"转录结果: {transcript}")

# 文字转语音
speech = ai.text_to_speech("你好，世界！")
with open("output.mp3", "wb") as f:
    f.write(speech)

实时音视频处理

Python - WebSocket 实时处理 
import asyncio
import websockets
import json
import base64
from typing import AsyncGenerator

class RealTimeAI:
    def __init__(self, api_key: str, ws_url: str = "wss://api.example.com/ws"):
        self.api_key = api_key
        self.ws_url = ws_url
    
    async def stream_audio_processing(self, audio_stream: AsyncGenerator) -> AsyncGenerator:
        """实时音频流处理"""
        async with websockets.connect(f"{self.ws_url}?token={self.api_key}") as websocket:
            # 发送配置
            await websocket.send(json.dumps({
                "type": "config",
                "model": "whisper-1",
                "language": "zh"
            }))
            
            async for audio_chunk in audio_stream:
                # 发送音频数据
                await websocket.send(json.dumps({
                    "type": "audio",
                    "data": base64.b64encode(audio_chunk).decode()
                }))
                
                # 接收转录结果
                response = await websocket.recv()
                result = json.loads(response)
                
                if result.get("type") == "transcription":
                    yield result["text"]
    
    async def stream_video_analysis(self, frame_stream: AsyncGenerator) -> AsyncGenerator:
        """实时视频流分析"""
        async with websockets.connect(f"{self.ws_url}?token={self.api_key}") as websocket:
            await websocket.send(json.dumps({
                "type": "config",
                "model": "gemini-pro-vision",
                "analysis_type": "object_detection"
            }))
            
            async for frame in frame_stream:
                await websocket.send(json.dumps({
                    "type": "frame",
                    "data": base64.b64encode(frame).decode()
                }))
                
                response = await websocket.recv()
                result = json.loads(response)
                
                if result.get("type") == "analysis":
                    yield result["objects"]

# 使用示例
async def main():
    rt_ai = RealTimeAI("your_api_key")
    
    # 模拟音频流
    async def mock_audio_stream():
        for i in range(10):
            yield b"mock_audio_data_" + str(i).encode()
            await asyncio.sleep(0.1)
    
    # 实时转录
    async for text in rt_ai.stream_audio_processing(mock_audio_stream()):
        print(f"转录: {text}")

if __name__ == "__main__":
    asyncio.run(main())

import requests
import json

url = "https://deepresearch2agi.cn/v1/chat/completions"
headers = {
    "Content-Type": "application/json",
    "Authorization": "Bearer YOUR_API_KEY"
}

    "model": "gpt-4o-mini",

    "messages": [
        {"role": "system", "content": "You are a helpful assistant."},
        {"role": "user", "content": "Hello!"}
    ]
}

result = response.json()

data["stream"] = True
response = requests.post(url, headers=headers, json=data, stream=True)

for line in response.iter_lines():
    if line:
        decoded = line.decode('utf-8')
        if decoded.startswith('data: '):
            if decoded[6:] != '[DONE]':
                chunk = json.loads(decoded[6:])
                if 'choices' in chunk and chunk['choices'][0]['delta'].get('content'):
                    print(chunk['choices'][0]['delta']['content'], end='')

Java - OkHttp

Java 
import okhttp3.*;
import org.json.JSONObject;
import org.json.JSONArray;

public class OpenAIClient {
    private static final String API_KEY = "YOUR_API_KEY";
    private static final String BASE_URL = "https://deepresearch2agi.cn/v1/chat/completions";

    public static void main(String[] args) throws Exception {
        OkHttpClient client = new OkHttpClient();

        JSONObject message1 = new JSONObject()
            .put("role", "system")
            .put("content", "You are a helpful assistant.");
        
        JSONObject message2 = new JSONObject()
            .put("role", "user")
            .put("content", "Hello!");

        JSONArray messages = new JSONArray()
            .put(message1)
            .put(message2);

        JSONObject requestBody = new JSONObject()
            .put("model", "gpt-4o-mini")
            .put("stream", false)
            .put("messages", messages);

        RequestBody body = RequestBody.create(
            requestBody.toString(),
            MediaType.parse("application/json")
        );

        Request request = new Request.Builder()
            .url(BASE_URL)
            .addHeader("Authorization", "Bearer " + API_KEY)
            .addHeader("Content-Type", "application/json")
            .post(body)
            .build();

        try (Response response = client.newCall(request).execute()) {
            String responseBody = response.body().string();
            JSONObject jsonResponse = new JSONObject(responseBody);
            String content = jsonResponse
                .getJSONArray("choices")
                .getJSONObject(0)
                .getJSONObject("message")
                .getString("content");
            System.out.println(content);
        }
    }
}

C# - HttpClient

C# 
using System;
using System.Net.Http;
using System.Text;
using System.Text.Json;
using System.Threading.Tasks;

class Program
{
    private static readonly string ApiKey = "YOUR_API_KEY";
    private static readonly string BaseUrl = "https://deepresearch2agi.cn/v1/chat/completions";

    static async Task Main(string[] args)
    {
        using var client = new HttpClient();
        client.DefaultRequestHeaders.Add("Authorization", $"Bearer {ApiKey}");

        var requestBody = new
        {
            model = "gpt-4o-mini",
            stream = false,
            messages = new[]
            {
                new { role = "system", content = "You are a helpful assistant." },
                new { role = "user", content = "Hello!" }
            }
        };

        var json = JsonSerializer.Serialize(requestBody);
        var content = new StringContent(json, Encoding.UTF8, "application/json");

        var response = await client.PostAsync(BaseUrl, content);
        var responseString = await response.Content.ReadAsStringAsync();
        
        using var doc = JsonDocument.Parse(responseString);
        var messageContent = doc.RootElement
            .GetProperty("choices")[0]
            .GetProperty("message")
            .GetProperty("content")
            .GetString();
        
        Console.WriteLine(messageContent);
    }
}

Go - net/http

Go 
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

const (
    apiKey  = "YOUR_API_KEY"
    baseURL = "https://deepresearch2agi.cn/v1/chat/completions"
)

type Message struct {
    Role    string `json:"role"`
    Content string `json:"content"`
}

type Request struct {
    Model    string    `json:"model"`
    Stream   bool      `json:"stream"`
    Messages []Message `json:"messages"`
}

func main() {
    reqBody := Request{
        Model:  "gpt-4o-mini",
        Stream: false,
        Messages: []Message{
            {Role: "system", Content: "You are a helpful assistant."},
            {Role: "user", Content: "Hello!"},
        },
    }

    jsonData, _ := json.Marshal(reqBody)
    
    req, _ := http.NewRequest("POST", baseURL, bytes.NewBuffer(jsonData))
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("Authorization", "Bearer "+apiKey)

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}

#include 

int main() {
    CURL *curl = curl_easy_init();
    if (curl) {
        struct curl_slist *headers = NULL;
        headers = curl_slist_append(headers, "Content-Type: application/json");
        headers = curl_slist_append(headers, "Authorization: Bearer YOUR_API_KEY");

        const char *payload = "{\"model\":\"gpt-4o-mini\",\"stream\":false,\"messages\":[{\"role\":\"system\",\"content\":\"You are a helpful assistant.\"},{\"role\":\"user\",\"content\":\"Hello!\"}]}";

        curl_easy_setopt(curl, CURLOPT_URL, "https://deepresearch2agi.cn/v1/chat/completions");
        curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);
        curl_easy_setopt(curl, CURLOPT_POSTFIELDS, payload);
        curl_easy_perform(curl);

        curl_slist_free_all(headers);
        curl_easy_cleanup(curl);
    }
    return 0;
}
